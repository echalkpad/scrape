    Suppose you have the following code:

    void InsertNode(tNode** node, int i){
   if(*node == NULL){
      *node = new tNode;
      (*node)->pLeft = NULL;
      (*node)->data = i;
      (*node)->pRight = NULL;
      SetRootNode(node);
      return;
   } else {
      if(i < (*node)->data)
         InsertNode(&((*node)->pLeft), i);
         if(i > (*node)->data)
            InsertNode(&((*node)->pRight), i);
         return;
   }
}

void Func(tNode **node){
   if(*node!=NULL){
      Func(&(*node)->pLeft);
      tNode *temp;
      temp = (*node)->pLeft;
      (*node)->pLeft= (*node)->pRight;
      (*node)->pRight = temp;
      Func(&(*node)->pRight);
   }
}

void traverse(tNode** nd){
   if(*nd!=NULL){
      traverse(&((*nd)->pLeft));
      traverse(&((*nd)->pRight));
      std::cout<<(*nd)->data<<std::endl;
   }
}
