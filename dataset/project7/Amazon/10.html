    Answer: First told him about Brute Force approach and then told him if we can iterate once and get the total sum

    int findIndex(A){

int sum =0;
   for(int i =0;i<A.lenght;i++)
   {
       sum+= A[i];
   }
   
   int lSum=0;
   for(int i=0;i<A.length;i++)
   {
       int rsum = sum - lsum-A[i];
       
       if(lsum==rsum)
       return i;
       
       lsum += A[i];
       
   }
   
   return -1;

}
