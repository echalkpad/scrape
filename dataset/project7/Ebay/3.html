    Program to calculate a pow n...

    I gave a answer with O(logN)

    public static long exp(int a , int n)
	{
		boolean is_N_Odd = false;
		if ( n % 2 > 0 )
		{
			n++;
			is_N_Odd =true;
		}

		if ( n == 2 )
			return a * a;
		else if (n == 1 )
			return a;
		else
		{
			long calc = exp (a , n/2) ;  
			if ( is_N_Odd )
				return calc * calc /a;
			else			
				return calc * calc ;
		}
	}
